\begin{frame}[t]{Results}{Threshold space: $\theta_p$ vs $\theta_r$ colored by score}

  \centering
  \includegraphics[width=0.85\textwidth,height=0.72\textheight,keepaspectratio]{assets/threshold_space_overview.png}

\begin{frame}[t]{Results}{Hysteresis margin analysis: diminishing returns}

  \centering
  \includegraphics[width=0.78\textwidth,height=0.68\textheight,keepaspectratio]{assets/margin_vs_best_overview.png}

  \vspace{0.05cm}
  \begin{itemize}
    \item Beyond $\theta_p - \theta_r \approx 50$~gCO$_2$eq/kWh: score plateaus or declines.
    \item Higher margins $\Rightarrow$ more idle time without proportional savings.
  \end{itemize}

\end{frame}

\subsection{Multi-Year Stability}

\begin{frame}[t]{Results}{Multi-year averaging attenuates peak savings}

  \centering
  \includegraphics[width=0.75\textwidth,height=0.55\textheight,keepaspectratio]{assets/multiyear_avg_score.png}

  \vspace{0.1cm}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{DE:} 43.4\% $\to$ 19.7\% \\
      \textbf{IT:} 32.7\% $\to$ 18.7\%
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{SE:} 23.2\% $\to$ 22.3\% \\
      \textbf{CN:} 6.2\% $\to$ 2.6\%
    \end{column}
  \end{columns}

  \vspace{0.1cm}
  \begin{block}{Practical implication}
    Multi-year optima may underestimate savings by up to $54\%$ due to smoothing.
  \end{block}

\end{frame}

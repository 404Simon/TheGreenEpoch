\section{Project Definition}
\subsection{Requirements}

\begin{frame}[t]{Motivation}{Why is it relevant and what's the goal?}

  \begin{itemize}
    \item LLM pretraining can use \emph{thousands of GPUs} over
      \emph{weeks to months}
      $\Rightarrow$ very high energy and CO$_2$ footprint.
    \item Grid CO$_2$ intensity varies strongly over time and region
      (renewable share, time of day, season).
    \item Idea: pause training during \emph{high-carbon} periods and resume in
      \emph{low-carbon} periods to reduce emissions.
    \item But: pausing/resuming introduces extra wall-clock time and
      checkpoint overhead (time and energy).
    \item Our simulation study aims to \textbf{quantify the trade-off}
      between
      \begin{itemize}
        \item reduced CO$_2$ emissions and
        \item increased training duration
      \end{itemize}
      in the \textbf{\emph{pretraining phase}}.
      \vspace{0.5cm}
    \item Citations:
      \begin{itemize}
        \item KIMI K2: Open Agentic Intelligence, 2023
        \item DeepSeek-V3 Technical Report, 2026
      \end{itemize}
  \end{itemize}

\end{frame}

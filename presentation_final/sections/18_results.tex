\subsection{Multi-Year Stability}

\begin{frame}[t]{Results}{Multi-year averaging attenuates peak savings}

  \centering
  \includegraphics[width=0.75\textwidth,height=0.62\textheight,keepaspectratio]{assets/multiyear_avg_score.png}

  \vspace{0.1cm}
  \begin{columns}[T]

      \textbf{IT:} 32.7\% $\to$ 18.7\%

    \begin{column}{0.48\textwidth}
      \textbf{SE:} 23.2\% $\to$ 22.3\% \\
      \textbf{CN:} 6.2\% $\to$ 2.6\%
    \end{column}
  \end{columns}

  \vspace{0.1cm}
  \begin{block}{Practical implication}
    Optimize on $\ge$3 years of data. Single-year optima may overestimate savings by $2\times$.
  \end{block}

\end{frame}

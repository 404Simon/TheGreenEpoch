\begin{frame}[t]{Outlook}{Future work}

  \textbf{Immediate extensions:}
  \begin{itemize}
    \item \textbf{Forecast-based policies:} Replace historical replay with
      predictive carbon intensity forecasts (e.g.\ using ECMWF or
      local grid operator data).
    \item \textbf{Geo-distributed training:} Combine temporal shifting with
      spatial flexibility for compounded savings (cf.\ Wiesner et al., 2026).
  \end{itemize}

  \vspace{0.3cm}
  \textbf{Longer-term:}
  \begin{itemize}
    \item \textbf{Embodied carbon awareness:} Ensure operational savings from
      pausing are not outweighed by amortized manufacturing emissions.
    \item \textbf{Multi-job scheduling:} Optimize across multiple training jobs
      sharing a GPU cluster.
  \end{itemize}

  \vspace{0.5cm}
  \begin{center}
    \textbf{All code and data are open-source:}\\
    \url{https://github.com/404simon/TheGreenEpoch}
  \end{center}

\end{frame}

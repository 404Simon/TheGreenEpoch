\subsection{Scoring of Policies}

\begin{frame}[t]{Scoring of Policies}{Key performance indicators}

  \begin{itemize}
    \item \textbf{Primary KPIs:}
      \begin{itemize}
        \item CO$_2$ savings relative to a baseline:
          \[
            S
            = \frac{GHG_{b} -
            GHG_{p}}{GHG_{b}} \times 100
          \]
        \item Time overhead relative to baseline:
          \[
            O
            = \frac{T_{p} -
            T_{b}}{T_{b}} \times 100
          \]
      \end{itemize}

      \vspace{0.2cm}
      \item \textbf{Composite score for ranking policies:}
      \[
        \text{score} = \frac{
          \alpha \cdot \frac{S}{100}
          + 1 - (1-\alpha) \cdot \frac{O}{B}
        }{2}
      \]
      where $B$ = overhead budget (200\%), $\alpha \in [0,1]$ controls
      the trade-off.
      \begin{itemize}
        \item $\alpha = 1$ (used in all experiments): score simplifies to
          $\frac{S}{200} + 0.5$.
          Savings maximization subject to overhead budget.
        \item $\alpha = 0$: score penalizes overhead alone.
      \end{itemize}

      \vspace{0.2cm}
      \item \textbf{Hard constraint:}
      \[
        O \leq B
      \]
      \begin{itemize}
        \item Only policies within the budget are considered.
        \item Among budget-feasible policies, the one with the
          highest score is selected.
      \end{itemize}
  \end{itemize}

\end{frame}
